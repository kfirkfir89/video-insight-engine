react-i18next, translations, formatting, and multi-language patterns.

## Setup

### DO ✅

```typescript
// lib/i18n.ts
import i18n from 'i18next';
import { initReactI18next } from 'react-i18next';
import LanguageDetector from 'i18next-browser-languagedetector';
import Backend from 'i18next-http-backend';

i18n
  .use(Backend)
  .use(LanguageDetector)
  .use(initReactI18next)
  .init({
    fallbackLng: 'en',
    supportedLngs: ['en', 'es', 'fr', 'de', 'ja'],
    
    debug: import.meta.env.DEV,
    
    interpolation: {
      escapeValue: false, // React already escapes
    },

    backend: {
      loadPath: '/locales/{{lng}}/{{ns}}.json',
    },

    detection: {
      order: ['localStorage', 'navigator', 'htmlTag'],
      caches: ['localStorage'],
    },

    ns: ['common', 'auth', 'errors'],
    defaultNS: 'common',
  });

export default i18n;
```

```tsx
// main.tsx
import './lib/i18n';
import { Suspense } from 'react';

function App() {
  return (
    <Suspense fallback={<LoadingSpinner />}>
      <Router />
    </Suspense>
  );
}
```

## Basic Usage

### DO ✅

```tsx
import { useTranslation } from 'react-i18next';

function HomePage() {
  const { t } = useTranslation();

  return (
    <div>
      <h1>{t('app.title')}</h1>
      <p>{t('app.description')}</p>
      
      <nav>
        <a href="/">{t('nav.home')}</a>
        <a href="/about">{t('nav.about')}</a>
      </nav>

      <button>{t('actions.save')}</button>
    </div>
  );
}

// With specific namespace
function LoginPage() {
  const { t } = useTranslation('auth');

  return (
    <form>
      <h1>{t('login.title')}</h1>
      
      <label>{t('login.email')}</label>
      <input type="email" />
      
      <label>{t('login.password')}</label>
      <input type="password" />
      
      <button type="submit">{t('login.submit')}</button>
    </form>
  );
}

// Multiple namespaces
function Dashboard() {
  const { t } = useTranslation(['common', 'dashboard']);

  return (
    <div>
      <h1>{t('dashboard:title')}</h1>
      <button>{t('common:actions.save')}</button>
    </div>
  );
}
```

```tsx
function UserGreeting({ user }: { user: User }) {
  const { t } = useTranslation();

  return (
    <div>
      {/* Simple interpolation */}
      <h1>{t('greeting', { name: user.name })}</h1>
      
      {/* Pluralization */}
      <p>{t('items.count', { count: user.cartItems })}</p>
      
      {/* Multiple values */}
      <p>{t('orderStatus', { orderId: '12345', status: 'shipped' })}</p>
    </div>
  );
}

// With React components (Trans)
import { Trans } from 'react-i18next';

function WelcomeMessage({ name }: { name: string }) {
  return (
    <Trans i18nKey="welcome" values={{ name }}>
      Welcome back, <strong>{{ name }}</strong>!
    </Trans>
  );
}
```

## Language Switcher

### DO ✅

```tsx
import { useTranslation } from 'react-i18next';

const languages = [
  { code: 'en', name: 'English', flag: '🇺🇸' },
  { code: 'es', name: 'Español', flag: '🇪🇸' },
  { code: 'fr', name: 'Français', flag: '🇫🇷' },
  { code: 'de', name: 'Deutsch', flag: '🇩🇪' },
  { code: 'ja', name: '日本語', flag: '🇯🇵' },
];

function LanguageSwitcher() {
  const { i18n } = useTranslation();

  const changeLanguage = (lng: string) => {
    i18n.changeLanguage(lng);
    // Update document direction for RTL languages
    document.documentElement.dir = ['ar', 'he'].includes(lng) ? 'rtl' : 'ltr';
    document.documentElement.lang = lng;
  };

  return (
    <select
      value={i18n.language}
      onChange={(e) => changeLanguage(e.target.value)}
      aria-label="Select language"
    >
      {languages.map((lang) => (
        <option key={lang.code} value={lang.code}>
          {lang.flag} {lang.name}
        </option>
      ))}
    </select>
  );
}

// Dropdown version
function LanguageDropdown() {
  const { i18n, t } = useTranslation();
  const [isOpen, setIsOpen] = useState(false);

  const currentLang = languages.find((l) => l.code === i18n.language);

  return (
    <div className="relative">
      <button
        onClick={() => setIsOpen(!isOpen)}
        className="flex items-center gap-2 px-3 py-2 border rounded"
        aria-expanded={isOpen}
        aria-haspopup="listbox"
      >
        <span>{currentLang?.flag}</span>
        <span>{currentLang?.name}</span>
        <ChevronDownIcon className="w-4 h-4" />
      </button>

      {isOpen && (
        <ul
          role="listbox"
          className="absolute top-full mt-1 w-full bg-white border rounded shadow-lg"
        >
          {languages.map((lang) => (
            <li
              key={lang.code}
              role="option"
              aria-selected={lang.code === i18n.language}
              onClick={() => {
                i18n.changeLanguage(lang.code);
                setIsOpen(false);
              }}
              className="flex items-center gap-2 px-3 py-2 hover:bg-gray-100 cursor-pointer"
            >
              <span>{lang.flag}</span>
              <span>{lang.name}</span>
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}
```

## Date & Number Formatting

### DO ✅

```tsx
import { useTranslation } from 'react-i18next';

function FormattedContent() {
  const { t, i18n } = useTranslation();

  // Date formatting
  const formatDate = (date: Date, options?: Intl.DateTimeFormatOptions) => {
    return new Intl.DateTimeFormat(i18n.language, {
      dateStyle: 'long',
      ...options,
    }).format(date);
  };

  // Number formatting
  const formatNumber = (num: number, options?: Intl.NumberFormatOptions) => {
    return new Intl.NumberFormat(i18n.language, options).format(num);
  };

  // Currency formatting
  const formatCurrency = (amount: number, currency = 'USD') => {
    return new Intl.NumberFormat(i18n.language, {
      style: 'currency',
      currency,
    }).format(amount);
  };

  // Relative time
  const formatRelativeTime = (date: Date) => {
    const rtf = new Intl.RelativeTimeFormat(i18n.language, { numeric: 'auto' });
    const diff = Math.round((date.getTime() - Date.now()) / (1000 * 60 * 60 * 24));
    return rtf.format(diff, 'day');
  };

  return (
    <div>
      <p>Date: {formatDate(new Date())}</p>
      <p>Number: {formatNumber(1234567.89)}</p>
      <p>Price: {formatCurrency(99.99)}</p>
      <p>Posted: {formatRelativeTime(new Date('2024-01-01'))}</p>
    </div>
  );
}

// Custom hook for formatting
function useFormatters() {
  const { i18n } = useTranslation();

  return {
    date: (date: Date, options?: Intl.DateTimeFormatOptions) =>
      new Intl.DateTimeFormat(i18n.language, options).format(date),

    number: (num: number, options?: Intl.NumberFormatOptions) =>
      new Intl.NumberFormat(i18n.language, options).format(num),

    currency: (amount: number, currency = 'USD') =>
      new Intl.NumberFormat(i18n.language, { style: 'currency', currency }).format(amount),

    percent: (value: number) =>
      new Intl.NumberFormat(i18n.language, { style: 'percent' }).format(value),

    list: (items: string[], type: 'conjunction' | 'disjunction' = 'conjunction') =>
      new Intl.ListFormat(i18n.language, { type }).format(items),
  };
}

// Usage
function PriceDisplay({ price }: { price: number }) {
  const { currency } = useFormatters();
  return <span>{currency(price)}</span>;
}
```

## RTL Support

### DO ✅

```tsx
// Detect and apply RTL
function RTLProvider({ children }: { children: React.ReactNode }) {
  const { i18n } = useTranslation();
  const isRTL = ['ar', 'he', 'fa', 'ur'].includes(i18n.language);

  useEffect(() => {
    document.documentElement.dir = isRTL ? 'rtl' : 'ltr';
    document.documentElement.lang = i18n.language;
  }, [i18n.language, isRTL]);

  return <div dir={isRTL ? 'rtl' : 'ltr'}>{children}</div>;
}

// RTL-aware styles (Tailwind)
function Card({ children }: { children: React.ReactNode }) {
  return (
    <div className="p-4 border rounded">
      {/* Use start/end instead of left/right */}
      <div className="ps-4 pe-2">  {/* padding-inline-start/end */}
        {children}
      </div>
      <button className="ms-auto">  {/* margin-inline-start */}
        Action
      </button>
    </div>
  );
}

//   border-inline-start: 1px solid;
```

## Lazy Loading Translations

### DO ✅

```typescript
// Load namespace on demand
function useNamespace(ns: string) {
  const { i18n } = useTranslation();

  useEffect(() => {
    if (!i18n.hasLoadedNamespace(ns)) {
      i18n.loadNamespaces(ns);
    }
  }, [i18n, ns]);
}

// Component with lazy-loaded translations
function AdminPanel() {
  const { t, ready } = useTranslation('admin', { useSuspense: false });

  if (!ready) return <LoadingSpinner />;

  return (
    <div>
      <h1>{t('title')}</h1>
      {/* ... */}
    </div>
  );
}
```

```typescript
// types/i18n.d.ts
import 'i18next';
import common from '../public/locales/en/common.json';
import auth from '../public/locales/en/auth.json';

declare module 'i18next' {
  interface CustomTypeOptions {
    defaultNS: 'common';

      common: typeof common;
      auth: typeof auth;

  }
}
```

## Quick Reference

| Hook/Function | Purpose |
|---------------|---------|
| `useTranslation()` | Get `t` function and `i18n` instance |
| `t('key')` | Translate a key |
| `t('key', { val })` | With interpolation |
| `i18n.changeLanguage()` | Switch language |
| `i18n.language` | Current language |
| `<Trans>` | Complex JSX translations |

| Intl API | Purpose |
|----------|---------|
| `DateTimeFormat` | Format dates |
| `NumberFormat` | Format numbers/currency |
| `RelativeTimeFormat` | "2 days ago" |
| `ListFormat` | "A, B, and C" |
| `PluralRules` | Plural forms |

| Pattern | Example |
|---------|---------|
| Nested keys | `t('nav.home')` |
| Namespace | `t('auth:login.title')` |
| Plurals | `t('items', { count: 5 })` |
| Interpolation | `t('hello', { name: 'John' })` |
